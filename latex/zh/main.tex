\begin{algorithm}[!t]
	\caption{隐藏事件生成}\label{alg_heg}
	\textbf{Input:}\\
	\hspace*{2em}Weighted temporal logical rules $\mathcal{R}$;\\
	\hspace*{2em}True events $\mathcal{O}$, Time set $\mathcal{T}$ of $\mathcal{O}$;\\
	\textbf{Output:}\\
	\hspace*{2em}Hidden events $\mathcal{H}$;
	\vspace{0.3em}
	\hrule
	\vspace{0.3em}
	\begin{algorithmic}[1]
		\STATE $\mathcal{C}\leftarrow\emptyset$;
		\FOR{$r\in \mathcal{R}$}
		\STATE $r_b=(p_0, p_1,\dots,p_{l-1}), p_h$ are body and head of $r$;
		\FOR{$p_i\in r_b$}
		\STATE $\mathcal{O}_{p_i}=\{\epsilon|\epsilon=(e_s,p,e_o,\tau)\in \mathcal{O},p=p_i\}$;
		\FOR{$\epsilon\in \mathcal{O}_{p_i}$}
		\STATE Construct path along $p_i$ with $\epsilon$ following time constraints, and record the min time of the last hop as $\tau_{min}$;
		\ENDFOR
		\ENDFOR
		\STATE With any complete body path with length $l$, use the head and tail entity of it and $p_h$ to construct a static triplet $(e_0,p_h,e_{l})$, then put it together with $\tau_{min}$ into set $\mathcal{C}$;
		\ENDFOR
		\STATE $\mathcal{C}_T\leftarrow\emptyset$;
		\FOR{$c=(e_s,p,e_o),\tau_{min}\in C$}
		\FOR{$\tau_i>\tau_{min}$ and $\tau_i\in\mathcal{T}$}
		\STATE Combine $c$ with $\tau_i$ to form a candidate hidden event $\epsilon=(e_s,p,e_o,\tau_i)$;
		\STATE Score $\epsilon$;
		\STATE Put $\epsilon$ with its score into $\mathcal{C}_T$;
		\ENDFOR
		\ENDFOR
		\STATE Sort $\mathcal{C}_T$ reversely by score and choose the top $N$ as $\mathcal{H}$.
	\end{algorithmic}
\end{algorithm}
